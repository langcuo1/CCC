一、变量的定义（var / let / const）

// 使用 var 声明（函数作用域，可重复声明）
var school = "光明小学";

// 使用 let 声明（块作用域，推荐）
let studentName = "小明";
let age = 18;

// 使用 const 声明（常量，不可重新赋值）
const PI = 3.14159;

// 变量重新赋值
age = 19;  // let 可以重新赋值
// PI = 3;  // ❌ const 不可重新赋值，会报错

console.log(school, studentName, age, PI);

二、变量类型（数据类型）

// 基本数据类型
let numInt    = 42;           // Number（整数）
let numFloat  = 3.14;         // Number（浮点数）
let str       = "Hello JS";   // String（字符串）
let bool      = true;         // Boolean（布尔）
let nothing   = null;         // Null（空值）
let notDef;                   // Undefined（未赋值）

// 引用数据类型
let arr       = [1, 2, 3];    // Array（数组）
let obj       = { name: "张三", score: 95 }; // Object（对象）

// 用 typeof 检测类型
console.log(typeof numInt);   // "number"
console.log(typeof str);      // "string"
console.log(typeof bool);     // "boolean"
console.log(typeof notDef);   // "undefined"
console.log(typeof arr);      // "object"
console.log(typeof obj);      // "object"
（点击运行查看结果）

三、表达式（运算符与表达式）

// 算术运算
let a = 10, b = 3;
console.log(a + b);   // 13  加
console.log(a - b);   // 7   减
console.log(a * b);   // 30  乘
console.log(a / b);   // 3.333...  除
console.log(a % b);   // 1   取余
console.log(a ** b);  // 1000  幂运算

// 比较运算（返回 Boolean）
console.log(5 == "5");    // true  (仅值相等)
console.log(5 === "5");   // false (值+类型均相等)
console.log(5 !== "5");   // true

// 逻辑运算
console.log(true && false); // false (与)
console.log(true || false); // true  (或)
console.log(!true);         // false (非)

// 字符串拼接
let greeting = "你好，" + "世界！";
console.log(greeting);       // 你好，世界！

// 三元表达式
let score = 85;
let grade = score >= 60 ? "及格" : "不及格";
console.log(grade);          // 及格

四、流程控制（条件 / 循环）

// if / else if / else
let hour = new Date().getHours();
if (hour < 12) {
    console.log("上午好！");
} else if (hour < 18) {
    console.log("下午好！");
} else {
    console.log("晚上好！");
}

// switch 语句
let day = new Date().getDay(); // 0=周日, 1=周一 ...
switch(day) {
    case 0: console.log("今天是周日"); break;
    case 6: console.log("今天是周六"); break;
    default: console.log("今天是工作日");
}

// for 循环（累加 1~100）
let sum = 0;
for (let i = 1; i <= 100; i++) {
    sum += i;
}
console.log("1到100的和：" + sum);  // 5050

// while 循环（猜数字逻辑）
let count = 0, n = 1;
while (n < 100) {
    n *= 2;
    count++;
}
console.log("2的幂次翻倍超过100需要：" + count + "次");

// for...of 遍历数组
let fruits = ["苹果", "香蕉", "橙子"];
for (let fruit of fruits) {
    console.log("水果：" + fruit);
}

五、函数（Function）

// 1. 函数声明
function greet(name) {
    return "你好，" + name + "！欢迎学习 JavaScript。";
}
console.log(greet("小红"));

// 2. 函数表达式
const square = function(x) {
    return x * x;
};
console.log("5的平方：" + square(5));

// 3. 箭头函数（ES6）
const add = (x, y) => x + y;
console.log("3+7=" + add(3, 7));

// 4. 默认参数
function power(base, exp = 2) {
    return Math.pow(base, exp);
}
console.log("2^10=" + power(2, 10));
console.log("5的平方（默认）=" + power(5));

// 5. 递归函数（阶乘）
function factorial(n) {
    if (n <= 1) return 1;
    return n * factorial(n - 1);
}
console.log("5! = " + factorial(5));   // 120

// 6. 高阶函数（回调）
function applyTwice(fn, value) {
    return fn(fn(value));
}
const double = x => x * 2;
console.log("对8连续乘以2两次：" + applyTwice(double, 8));  // 32
（点击运行查看结果）
