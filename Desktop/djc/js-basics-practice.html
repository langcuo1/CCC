1 变量的定义

// 三种变量声明方式：var、let、const
var  oldStyle  = "我是 var 声明的变量（函数作用域）";
let  blockVar  = "我是 let 声明的变量（块级作用域）";
const PI       = 3.14159;          // const 声明常量，不可重新赋值

// 重新赋值演示
let score = 60;
score = 85;                        // let 可以重新赋值

// 输出
console.log(oldStyle);
console.log(blockVar);
console.log("PI =", PI);
console.log("score 重赋值后：", score);

2 变量类型

// JavaScript 有 7 种基本数据类型
let numInt    = 42;                // Number  整数
let numFloat  = 3.14;             // Number  浮点数
let str       = "Hello, World!";  // String  字符串
let flag      = true;             // Boolean 布尔值
let nothing   = null;             // Null    空值
let undef;                        // Undefined 未定义
let sym       = Symbol("id");     // Symbol  唯一标识

// 引用类型
let arr  = [1, 2, 3];            // Array   数组
let obj  = { name: "张三", age: 20 }; // Object  对象

// typeof 运算符检测类型
console.log("numInt   typeof:", typeof numInt);
console.log("str      typeof:", typeof str);
console.log("flag     typeof:", typeof flag);
console.log("nothing  typeof:", typeof nothing);  // 历史遗留：输出 "object"
console.log("undef    typeof:", typeof undef);
console.log("sym      typeof:", typeof sym);
console.log("arr      typeof:", typeof arr);
console.log("obj      typeof:", typeof obj);

// 类型转换
let numStr = "100";
console.log("字符串转数字：", Number(numStr) + 1);   // 101
console.log("数字转字符串：", String(numInt));         // "42"
console.log("转布尔值(0)：",  Boolean(0));             // false
console.log("转布尔值(1)：",  Boolean(1));             // true
▶ 运行输出

3 表达式

// 算术运算符
let a = 10, b = 3;
console.log("a + b =", a + b);   // 13
console.log("a - b =", a - b);   // 7
console.log("a * b =", a * b);   // 30
console.log("a / b =", a / b);   // 3.333...
console.log("a % b =", a % b);   // 1  取余
console.log("a ** b =", a ** b); // 1000 幂运算

// 自增 / 自减
let x = 5;
console.log("x++ =", x++);  // 先用后增：5
console.log("++x =", ++x);  // 先增后用：7

// 比较运算符 → 返回 Boolean
console.log("10 == '10'  :", 10 == "10");   // true  (宽松相等)
console.log("10 === '10' :", 10 === "10");  // false (严格相等)
console.log("10 !== 3    :", 10 !== 3);     // true

// 逻辑运算符
let t = true, f = false;
console.log("t && f :", t && f);  // false
console.log("t || f :", t || f);  // true
console.log("!t     :", !t);      // false

// 三元表达式
let age = 18;
let status = (age >= 18) ? "成年人" : "未成年人";
console.log("age 18 →", status);

// 字符串拼接 & 模板字面量
let name = "小明";
console.log("你好，" + name + "！");          // 传统拼接
console.log(`你好，${name}！你今年 ${age} 岁。`); // 模板字面量
▶ 运行输出

4 流程控制

// ----- if / else if / else -----
let grade = 78;
let level;
if (grade >= 90) {
    level = "优秀";
} else if (grade >= 75) {
    level = "良好";
} else if (grade >= 60) {
    level = "及格";
} else {
    level = "不及格";
}
console.log(`成绩 ${grade} → 等级：${level}`);

// ----- switch -----
let day = new Date().getDay(); // 0=周日, 1=周一, ...
let dayName;
switch (day) {
    case 0: dayName = "周日"; break;
    case 1: dayName = "周一"; break;
    case 2: dayName = "周二"; break;
    case 3: dayName = "周三"; break;
    case 4: dayName = "周四"; break;
    case 5: dayName = "周五"; break;
    case 6: dayName = "周六"; break;
    default: dayName = "未知";
}
console.log("今天是：" + dayName);

// ----- for 循环 -----
let sum = 0;
for (let i = 1; i <= 10; i++) {
    sum += i;
}
console.log("1+2+...+10 =", sum);

// ----- while 循环 -----
let n = 1, factorial = 1;
while (n <= 5) {
    factorial *= n;
    n++;
}
console.log("5! =", factorial);

// ----- for...of 遍历数组 -----
let fruits = ["苹果", "香蕉", "橙子"];
let fruitStr = "";
for (let fruit of fruits) {
    fruitStr += fruit + " ";
}
console.log("水果列表：", fruitStr.trim());

// ----- break / continue -----
let result = "";
for (let i = 0; i < 10; i++) {
    if (i === 3) { result += "(跳过3) "; continue; }
    if (i === 7) { result += "(在7处停止)"; break; }
    result += i + " ";
}
console.log(result);
▶ 运行输出

5 函数

// ----- 函数声明 -----
function greet(name) {
    return `你好，${name}！欢迎学习 JavaScript。`;
}
console.log(greet("同学"));

// ----- 函数表达式 -----
const add = function(a, b) {
    return a + b;
};
console.log("add(3, 5) =", add(3, 5));

// ----- 箭头函数 -----
const multiply = (a, b) => a * b;
console.log("multiply(4, 6) =", multiply(4, 6));

// ----- 默认参数 -----
function power(base, exponent = 2) {
    return Math.pow(base, exponent);
}
console.log("power(3)    =", power(3));      // 9
console.log("power(2, 8) =", power(2, 8));   // 256

// ----- 递归函数 -----
function fibonacci(n) {
    if (n <= 1) return n;
    return fibonacci(n - 1) + fibonacci(n - 2);
}
let fibSeq = "";
for (let i = 0; i < 8; i++) fibSeq += fibonacci(i) + " ";
console.log("Fibonacci(0-7):", fibSeq.trim());

// ----- 函数作为参数（高阶函数）-----
function applyOperation(a, b, fn) {
    return fn(a, b);
}
console.log("高阶函数 applyOperation(10, 4, 相减):", applyOperation(10, 4, (a, b) => a - b));

// ----- 立即执行函数 (IIFE) -----
const circleArea = (function(r) {
    return (Math.PI * r * r).toFixed(2);
})(5);
console.log("半径5的圆面积:", circleArea);
▶ 运行输出
